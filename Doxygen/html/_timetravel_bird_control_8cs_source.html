67  protected override void OnRender(DrawingContext drawingContext)
68  {
69  if (this.Renderer != null)
70  {
71  if (!this.Logic.InTimeTravel)
72  {
73  drawingContext.DrawDrawing(this.Renderer.BuildGameDrawing());
74  this.Renderer.BuildText(drawingContext);
75  this.Renderer.DrawWindEffect(drawingContext, this.gifDir["wind"][this.windFrame]);
76  this.Renderer.DrawWeather(drawingContext, this.gifDir["weather"][this.weatherFrame]);
77  if (this.Logic.Launch == false)
78  {
79  this.Renderer.DrawBird(drawingContext, this.gifDir[this.currentBird][this.birdGifFrame]);
80  }
81 
82  _ = this.birdGifFrame == this.birdMaxFrame - 1 ? this.birdGifFrame = 0 : this.birdGifFrame++;
83  _ = this.windFrame == 4 ? this.windFrame = 0 : this.windFrame++;
84  _ = this.weatherFrame == 10 ? this.weatherFrame = 0 : this.weatherFrame++;
85  }
86  else
87  {
88  if (!this.Logic.EndGameAnimation || !this.Logic.InTimeTravel)
89  {
90  this.Renderer.DrawTimeTravel(drawingContext, this.gifDir["travel"][this.timeTravelFrame++]);
91  if (this.timeTravelFrame == 48)
92  {
93  this.timeTravelFrame = 0;
94  this.Logic.InTimeTravel = false;
95  }
96  }
97  else
98  {
99  this.MainTimer.Stop();
100  this.Renderer.Reset();
101  this.Logic.BirdReset();
102  }
103  }
