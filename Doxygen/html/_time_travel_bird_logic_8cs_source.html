236  public void OneTick()
237  {
238  if (!this.EndGameAnimation)
239  {
240  if (this.Model.Bird.RealY < 500)
241  {
242  this.BirdCityTick();
243  }
244  else
245  {
246  this.BirdSkyTick();
247  }
248 
249  this.BackgroundTick();
250 
251  // this.LauncherTick();
252  if (this.Model.Bird.IsCollision(this.Model.Weather))
253  {
254  this.Model.Bird.DX += (random.NextDouble() * random.Next(-5, 2)) * (1 - (this.Model.Bird.WeatherResistance / 100));
255  this.Model.Bird.Rad += (Math.PI / 60) * (1 - (this.Model.Bird.WeatherResistance / 100));
256  }
257 
258  if (this.Model.Bird.IsCollision(this.Model.Wind))
259  {
260  this.Model.Bird.DX += (random.NextDouble() * random.Next(-3, 0)) * (1 - (this.Model.Bird.WeatherResistance / 100));
261  this.Model.Bird.Rad += (Math.PI / 180) * (1 - (this.Model.Bird.WeatherResistance / 100));
262  }
263 
264  if (this.Model.Bird.RealY > -30)
265  {
266  if (this.Model.HighestMeter < this.Model.Bird.RealX)
267  {
268  this.Model.HighestMeter = this.Model.Bird.RealX;
269  }
270  }
271 
272  if (!this.IsThereWeather)
273  {
274  if (this.WeatherTickCounter < 10)
275  {
276  this.WeatherTickCounter++;
277  }
278  else
279  {
280  this.WeatherTickCounter = 0;
281  this.WeatherGenerateTick();
282  }
283  }
284  else
285  {
286  this.WeatherTick();
287  }
288 
289  if (!this.IsThereWind)
290  {
291  if (this.WindTickCounter < 10)
292  {
293  this.WindTickCounter++;
294  }
295  else
296  {
297  this.WindTickCounter = 0;
298  this.WindGenerateTick();
299  }
300  }
301  else
302  {
303  this.WindTick();
304  }
305  }
306  else
307  {
308  this.Model.Bird.ItemCoordX += 20;
309  if (this.Model.Bird.ItemCoordX > 1280)
310  {
311  this.InTimeTravel = true;
312  }
313  }
314 
315  if (this.IsGameFinished())
316  {
317  this.InTimeTravel = false;
318  this.EndGameAnimation = false;
319  this.GameIsOver = true;
320  this.BirdReset();
321  //this.Model.Bird.RealX = 10;
322  DateTime now = DateTime.Now;
323  TimeSpan played = now.Subtract(this.Model.GameStartingTime);
324  string playedInString = $"{played.Days}days {played.Hours}:{played.Minutes}:{played.Seconds}";
325  GameIO.SaveHighScore(playedInString, this.Model.Launches);
326  GameIO.DeleteSavedGame(this.Model.LoadedGameName);
327  this.GameFinished?.Invoke(this, EventArgs.Empty);
328  }
329  }
